# movie_bot
Бот для выбора фильма с друзьями
import random
import re

class SmartChatBot:
    def __init__(self):
        self.responses = {
            r'привет': ['Привет!', 'Здравствуйте', 'Привет! Как я могу помочь?'],
            r'фильм': ['Что хотим почувствовать? : А) Хочу плакать…; Б) У меня хорошее настроение, я хочу смеяться!!!;
              В) Детективчик или триллер звучит как план;'],
            r'а\)': ['Выберите степень грусти: Г) Я готов превратиться в лужу; Д) Просто немного грустно; 
              Е) Что-то среднее;'],
            r'г\)|д\)|е\)': ['Вы хотите изменить свое состояние? 1) ДА; 2) Нет, я хочу погрузиться в это еще сильнее;
              3) Просто что-то такое'],
            r'б\)': ['К чему сейчас сильнее тяга? Ж) Черный юмор мне необходим; З) Я приличный человек, но посмеяться хочу;               И) Сама воспитанность, умеренное хихихи;'],
            r'в\)': ['Насколько сильно должно быть страшно? К) Очень, прям очень!; Л) Я скорее люблю подумать;
              М) Всего должно быть в меру);'],
            r'ж\)|з\)|и\)': ['Вы будете смотреть одни? 4) Да; 5) Нет, с друзьями;
              6) Нет, с родителями или приличными друзьями;'],
            r'к\)|л\)|м\)': ['Хотите быстро уснуть? 7) Да; 8) Нет; 9) Меня ничего не берет, в любом случае усну;'],
            r'1': ['Наше предложение: Правила съема: Метод Хитча'],
            r'2': ['Наше предложение: Хатико'],
            r'3': ['Наше предложение: Привидение'],
            r'4': ['Наше предложение: Без чувств'],
            r'5': ['Наше предложение:Маска'],
            r'6': ['Наше предложение: Отель "Гранд Будапешт"'],
            r'7': ['Наше предложение: Достать ножи первая часть'],
            r'8': ['Наше предложение: Невидимый гость'],
            r'9': ['Наше предложение: Семь'],
            r'спасибо|спасибо большое': ['Без проблем! Хорошего просмотра!', 'Обращайтесь:)'],
            r'пока|до свидания': ['До встречи! Хорошего просмотра!']
        }

    def get_response(self, text):
        text = text.lower()
        
        # Обычная обработка сообщений
        for pattern, responses in self.responses.items():
            if re.search(pattern, text):
                return random.choice(responses)
        
        return "Извините, я не понял ваш запрос. Напишите 'фильм', чтобы я мог подобрать для вас кино!"

bot = SmartChatBot()

print("Бот: Приветствую! Я ваш помощник по подбору фильмов.")
name = input("Введите ваше имя: ")
print(f"Бот: Добро пожаловать, {name}! Хотите подобрать фильм? Напишите 'фильм'.")

while True:
    request = input(f"{name}: ")
    if request.lower() in ['пока', 'до свидания', 'выход', 'спасибо']:
        print('Бот: До свидания! Хорошего просмотра!')
        break
    else:
        response = bot.get_response(request)
        print('Бот:', response)
